Sección 13 de 25 - Scope de PHP

              $outside_variable = "Esta es una variable externa";
              
              function my_function() {
                  global $outside_variable;
                  echo $outside_variable;
              }
              
              my_function();
              echo "<pre>";
              print_r($GLOBALS);
              
              function my_function2() {
                  echo $GLOBALS["outside_variable"];
              }
              
              my_function2();

Sección 21 de 25 - Funciones Anonimas

                    
              <?php

              $numbers = [1,2,3,4];
              
              $numbers_by_2 = array_map(function($current) {
                  return $current * 2;
              }, $numbers);
              
              echo "<pre>";
              
              
              var_dump($numbers_by_2);
              
              
              $multiplicar_por_3 = function($current) {
                  return $current * 3;
              };
              
              $numbers_by_3 = array_map($multiplicar_por_3, $numbers_by_2);
              
              echo "<pre>";
              
              
              var_dump($numbers_by_3);
              
              $michi = "Mr. Michi";
              
              $view_michi_name = function() use ($michi) {
                  echo $michi;
              };
              
              $view_michi_name();

Sección 21 de 25 - arrow functions

                    
              <?php

              $cajero = 10;
              
              $add_cajero = fn($add) => $cajero + $add;
              
              echo $add_cajero(20);
              echo "<br>";
              
              # Esto no fnciona... las variables globales no se pueden modificar, solo leer
              $add_cajero2 = fn($add) => $cajero += $add;
              
              $add_cajero2(2);
              $add_cajero2(20);
              $add_cajero2(8);
              
              echo $cajero;
              
              $edades = [5,21,50,9,18];
              
              $mayores_de_edad = array_filter($edades, fn($current) => $current >= 18);
              echo "<pre>";
              print_r($mayores_de_edad);
